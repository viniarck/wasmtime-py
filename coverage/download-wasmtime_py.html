1# Helper script to download a precompiled binary of the wasmtime dll for the 

2# current platform. Currently always downloads the dev release of wasmtime. 

3 

4import io 

5import os 

6import platform 

7import shutil 

8import sys 

9import tarfile 

10import urllib.request 

11import zipfile 

12 

13 

14def main(platform, arch): 

15 is_zip = False 

16 if arch == 'AMD64': 

17 arch = 'x86_64' 

18 if platform == 'linux': 

19 filename = 'wasmtime-dev-{}-linux-c-api.tar.xz'.format(arch) 

20 libname = '_libwasmtime.so' 

21 elif platform == 'win32': 

22 filename = 'wasmtime-dev-{}-windows-c-api.zip'.format(arch) 

23 is_zip = True 

24 libname = '_wasmtime.dll' 

25 elif platform == 'darwin': 

26 filename = 'wasmtime-dev-{}-macos-c-api.tar.xz'.format(arch) 

27 libname = '_libwasmtime.dylib' 

28 else: 

29 raise RuntimeError("unknown platform: " + sys.platform) 

30 

31 url = 'https://github.com/bytecodealliance/wasmtime/releases/download/dev/' 

32 url += filename 

33 print('Download', url) 

34 dirname = '{}-{}'.format(platform, arch) 

35 dst = os.path.join('wasmtime', dirname, libname) 

36 try: 

37 shutil.rmtree(os.path.dirname(dst)) 

38 except Exception: 

39 pass 

40 try: 

41 shutil.rmtree(os.path.join('wasmtime', 'include')) 

42 except Exception: 

43 pass 

44 os.makedirs(os.path.dirname(dst)) 

45 os.makedirs(os.path.join('wasmtime', 'include')) 

46 

47 with urllib.request.urlopen(url) as f: 

48 contents = f.read() 

49 

50 def final_loc(name): 

51 if name.endswith('.h'): 

52 return os.path.join('wasmtime', 'include', os.path.basename(name)) 

53 elif name.endswith('.dll') or name.endswith('.so') or name.endswith('.dylib'): 

54 return dst 

55 else: 

56 return None 

57 

58 if is_zip: 

59 t = zipfile.ZipFile(io.BytesIO(contents)) 

60 for member in t.namelist(): 

61 loc = final_loc(member) 

62 if not loc: 

63 continue 

64 contents = t.read(member) 

65 with open(loc, "wb") as f: 

66 f.write(contents) 

67 else: 

68 t = tarfile.open(fileobj=io.BytesIO(contents)) 

69 for member in t.getmembers(): 

70 loc = final_loc(member.name) 

71 if not loc: 

72 continue 

73 contents = t.extractfile(member).read() 

74 with open(loc, "wb") as f: 

75 f.write(contents) 

76 

77 if not os.path.exists(dst): 

78 raise RuntimeError("failed to find dynamic library") 

79 

80 

81if __name__ == '__main__': 

82 if len(sys.argv) > 2: 

83 main(sys.argv[1], sys.argv[2]) 

84 else: 

85 main(sys.platform, platform.machine())
